#version 330 core

in vec4 out_color;
in vec2 out_texCoords;
in float out_textureMode;

uniform sampler2D myTexture;

void main() 
{
	if(out_textureMode == 0) {
		gl_FragColor = out_color;
	}
	if(out_textureMode == 1) {
		gl_FragColor = texture(myTexture, out_texCoords);
		if(gl_FragColor.a < 0.5) {
			discard;
		}
	}
}
